Validasi berbeda tergantung tipe data. Tentukan tipe datamu dulu.
Kenali tipe datamu
TipeDefinisiKey unikContoh

def run_quality_check(df: pd.DataFrame, cfg: dict) -> dict:
    """Jalankan cek quality dasar, kembalikan ringkasan."""
    results = {
        'row_count'      : len(df),
        'duplicate_rows' : int(df.duplicated().sum()),
        'missing_pct'    : (df.isnull().sum() / len(df) * 100).round(2).to_dict(),
        'row_count_ok'   : len(df) >= cfg.get('min_rows', 100),
    }
    return results

qc = run_quality_check(df_raw, cfg['quality'])
if not qc['row_count_ok']:
    raise ValueError(f"Data terlalu sedikit: {qc['row_count']} baris")

Cek wajib — Panel data
⚠️
Kombinasi (entitas × waktu) harus unik
Panel data punya dua dimensi sekaligus: entitas (user, toko) dan waktu. Duplikat pada kombinasi keduanya adalah tanda data rusak — bukan sekadar duplikat baris biasa.
key_cols = ['user_id', 'bulan']   # sesuaikan dengan datamu

# 1. Cek duplikat pada kombinasi key (entitas + waktu)
n_dup = df.duplicated(subset=key_cols).sum()
print(f"Duplikat ({', '.join(key_cols)}): {n_dup}")
assert n_dup == 0, "Panel data tidak boleh ada duplikat pada kombinasi (entitas, waktu)"

# 2. Cek kelengkapan panel (balanced vs unbalanced)
n_entities = df['user_id'].nunique()
n_periods  = df['bulan'].nunique()
n_expected = n_entities * n_periods
print(f"Baris actual: {len(df):,} | Expected jika balanced: {n_expected:,}")
if len(df) < n_expected:
    print(f"⚠ Unbalanced panel: {n_expected - len(df):,} kombinasi hilang")

# 3. Cek konsistensi kehadiran tiap entitas
coverage = df.groupby('user_id')['bulan'].nunique()
print(f"Periode per user — min: {coverage.min()} | max: {coverage.max()} | median: {coverage.median()}")
💡
Balanced atau unbalanced — keduanya valid, tapi harus terdokumentasi
Balanced = semua entitas hadir di semua periode. Unbalanced = ada yang absen di beberapa periode. Catat mana yang berlaku di notebook dan pertimbangkan imputasi atau masking saat modelling.
Standar Evaluasi Model
Strategi split dan metrik evaluasi berbeda per tipe data. Tentukan keduanya di config sebelum mulai training.
Pilih strategi split
❌
Jangan random split untuk time series dan panel data
Random split menyebabkan data leakage: model "melihat masa depan" saat training. Test set harus selalu periode yang lebih baru.
Tipe dataStrategi splitCatatan
Cross-sectionaltrain_test_split(stratify=y)Wajib stratify jika imbalanced
Time seriesExpanding window (TimeSeriesSplit)Jangan random split!
Panel dataSplit berdasarkan periodeJangan split per baris!
ClusteringTidak ada splitEvaluasi internal + bisnis
Metrik per task
Regresi
MAE  = mean_absolute_error(y_test, y_pred)   # error rata2 dalam unit asli
RMSE = mean_squared_error(y_test, y_pred)**0.5  # sensitif ke outlier
MAPE = np.mean(np.abs((y_test - y_pred) / (y_test + 1e-8))) * 100  # % error
Klasifikasi
auc = roc_auc_score(y_test, y_pred_proba)     # default terbaik kebanyakan kasus

Clustering
💡
Tidak ada "benar" atau "salah" — validasi bisnis wajib
Metrik internal hanya membantu memilih jumlah cluster yang stabil secara statistik. Cluster yang bagus secara statistik belum tentu bermakna bisnis — wajib dikonfirmasi dengan domain expert atau klien.
from sklearn.metrics import silhouette_score, davies_bouldin_score
from sklearn.cluster import KMeans
import matplotlib.pyplot as plt

# 1. Elbow method — cari "siku" di plot WCSS untuk pilih k
wcss = []
for k in range(2, 11):
    km = KMeans(n_clusters=k, random_state=42, n_init=10)
    km.fit(X_scaled)
    wcss.append(km.inertia_)
plt.plot(range(2, 11), wcss, 'o-')
plt.xlabel('Jumlah cluster (k)'); plt.ylabel('WCSS'); plt.title('Elbow Method')

# 2. Silhouette Score — semakin tinggi semakin baik (rentang -1 s.d. 1)
sil = silhouette_score(X_scaled, labels)
print(f"Silhouette Score : {sil:.3f}   # patokan awal: > 0.3")

# 3. Davies–Bouldin Index — semakin rendah semakin baik
dbi = davies_bouldin_score(X_scaled, labels)
print(f"Davies–Bouldin   : {dbi:.3f}   # patokan awal: < 1.0")

# 4. WAJIB: profil tiap cluster harus bisa dijelaskan ke klien
df['cluster'] = labels
print(df.groupby('cluster')[['fitur_utama_1', 'fitur_utama_2']].mean().round(2))
✅
Pilih k yang masuk akal bisnis, bukan hanya yang paling bagus metriknya
Jika k=7 menghasilkan Silhouette tertinggi tapi tim tidak bisa membedakan dan menamai 7 segmen dengan jelas — pilih k yang lebih kecil. Cluster harus bisa ditindaklanjuti oleh bisnis.

Metadata tiap run — meta_[run_id].json
Catatan kondisi saat model dijalankan. Berguna untuk audit dan debug: "kenapa prediksi bulan ini berbeda dari bulan lalu?" — jawaban biasanya ada di file ini.
import json
from datetime import datetime

def save_model_metadata(run_id: str, params: dict, metrics: dict,
                        feature_cols: list, model_path, cfg: dict) -> None:
    """Simpan metadata run ke JSON — dipanggil setelah setiap training."""
    meta = {
        'run_id'       : run_id,                      # '20250315_1430'
        'trained_at'   : datetime.now().isoformat(),  # timestamp lengkap ISO
        'n_features'   : len(feature_cols),
        'feature_cols' : feature_cols,
        'model_params' : params,                      # hyperparameter yang dipakai
        'metrics'      : metrics,                     # MAE, AUC, dll.
        'train_period' : {
            'start': str(cfg['data']['date_start']),
            'end'  : str(cfg['data']['date_end']),
        },
    }
    out = model_path / f'meta_{run_id}.json'
    with open(out, 'w') as f:
        json.dump(meta, f, indent=2, ensure_ascii=False)
    print(f"✓ Metadata tersimpan: {out.name}")

def log_experiment(params: dict, metrics: dict, notes: str = '') -> None:
    """Catat satu percobaan model ke experiments.csv."""
    row = {
        'timestamp'  : pd.Timestamp.now().isoformat(),
        'run_id'     : datetime.now().strftime('%Y%m%d_%H%M'),
        'model_type' : params.get('model_type', ''),
        'n_features' : params.get('n_features', ''),
        'train_score': metrics.get('train_score', ''),
        'test_score' : metrics.get('test_score', ''),
        'MAE'        : metrics.get('MAE', ''),
        'notes'      : notes,
    }
    exp_file = REPORT_PATH / 'experiments.csv'
    pd.DataFrame([row]).to_csv(exp_file, mode='a',
        header=not exp_file.exists(), index=False)

Pipeline Maturity Levels
Tidak semua project butuh pipeline otomatis penuh. Pilih level yang sesuai.
Empat level pipeline
Level 1 — Manual (one-time)
Notebook dijalankan sekali oleh DS. Output dikirim langsung ke klien. Cocok: analisis ad-hoc, proof-of-concept.
Level 2 — Semi-otomatis (berkala manual)
DS menjalankan pipeline secara manual setiap periode. Ada run_pipeline.py tapi tidak terjadwal. Cocok: reporting bulanan.
Level 3 — Terjadwal (DE-managed)
DE menjadwalkan run_pipeline.py via cron atau Cloud Scheduler. DS tidak perlu terlibat setiap run. Cocok: dashboard live harian/mingguan.
Level 4 — Production (DE-owned)
Pipeline sepenuhnya dikelola DE dengan monitoring, alerting, dan retry logic di Airflow / Cloud Composer. Cocok: produk kritikal, SLA ketat.
Cara tentukan level yang tepat

Pipeline yang gagal diam-diam lebih berbahaya dari pipeline yang crash dengan jelas.
Setup logging standar
import logging
from pathlib import Path

def setup_logger(name: str, log_file: Path = None) -> logging.Logger:
    """Setup logger dengan output ke console dan opsional ke file."""
    logger = logging.getLogger(name)
    logger.setLevel(logging.INFO)
    fmt = logging.Formatter('%(asctime)s | %(levelname)s | %(name)s | %(message)s')
    ch = logging.StreamHandler()
    ch.setFormatter(fmt)
    logger.addHandler(ch)
    if log_file:
        log_file.parent.mkdir(parents=True, exist_ok=True)
        fh = logging.FileHandler(log_file)
        fh.setFormatter(fmt)
        logger.addHandler(fh)
    return logger

logger = setup_logger(__name__, log_file=Path('outputs/reports/pipeline.log'))
Level logging yang tepat
logger.debug(...)    # detail teknis — nonaktif di produksi
logger.info(...)     # status normal: "Step 1 selesai, 50k baris"
logger.warning(...)  # tidak normal tapi pipeline tetap jalan
logger.error(...)    # operasi gagal — selalu diikuti raise

def validate_dashboard_output(df: pd.DataFrame, cfg: dict) -> None:
    """Validasi sebelum upload ke GCS/BQ untuk dashboard."""
    required_cols = cfg.get('dashboard_required_cols', [])
    missing = [c for c in required_cols if c not in df.columns]
    assert not missing, f"Kolom hilang: {missing}"
    assert len(df) >= cfg.get('min_dashboard_rows', 100), f"Output terlalu sedikit: {len(df)}"
    for col in required_cols:
        assert df[col].isnull().sum() == 0, f"Missing value di {col}"
    print(f"✓ Dashboard valid: {len(df):,} baris")
